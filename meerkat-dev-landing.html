Integrate in minutes

REST API, TypeScript SDK, MCP server, or one-click OpenClaw skill.

terminal 
# Shield: scan incoming content
curl -X POST https://api.meerkat.ai/v1/shield \
  -H "Authorization: Bearer $MEERKAT_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{"input": "email body or web content here"}'

# Verify: check before executing
curl -X POST https://api.meerkat.ai/v1/verify \
  -H "Authorization: Bearer $MEERKAT_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{
    "input": "user instruction",
    "output": "what the AI is about to do",
    "context": "source data",
    "domain": "healthcare"
  }'
# Install from ClawHub
clawhub install meerkat-governance

import { MeerkatInterceptor } from '@meerkat-ai/openclaw';

const meerkat = new MeerkatInterceptor({
  apiKey: process.env.MEERKAT_API_KEY,
  domain: 'healthcare',
});

// Before processing untrusted content
const shield = await meerkat.shield(emailBody, {
  source: 'email'
});
if (!shield.safe) return; // blocked

// Before executing an action
const result = await meerkat.interceptToolCall({
  toolName: 'send_email',
  userIntent: 'Reply to the contract summary',
  actionDescription: 'Sending email with revenue split',
  sourceContext: contractText,
});

if (!result.allowed) {
  console.log(result.message);
  // "BLOCKED: numbers reversed (60/40 vs 40/60)"
}
